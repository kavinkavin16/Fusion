SELECT  
HP.PARTY_NAME,
HP.PARTY_NUMBER,
HPS.PARTY_SITE_NAME,
PPNF.FULL_NAME AS Buyer,
PHA.Segment1,
PHA.APPROVED_DATE,
PLA.PRC_BU_ID,
PLA.REQ_BU_ID ,
PLA.ITEM_ID,
PLA.LINE_NUM ,
PLA.QUANTITY ,
PLA.LIST_PRICE_PER_UNIT,
PLA.UNIT_PRICE,
PLLA.NEED_BY_DATE ,
PLLA.PROMISED_DATE ,
PLLA.REQUESTED_SHIP_DATE,
PLLA.PROMISED_SHIP_DATE,
PLLA.CLOSED_FOR_RECEIVING_DATE ,
PLLA.QUANTITY_RECEIVED,
PLLA.QUANTITY_ACCEPTED,
PLLA.QUANTITY_REJECTED,
PDA.QUANTITY_CANCELLED ,
PDA.DESTINATION_SUBINVENTORY ,
PDA.QUANTITY_BILLED ,
PDA.AMOUNT_BILLED,
RSH.RECEIPT_NUM,
ESIB.ITEM_NUMBER,
PS.SEGMENT1 AS SEGMENT_1,
HOU.NAME AS PURCHASING_BU,
HOU.NAME AS REQUESTING_BU 
FROM HZ_PARTIES HP,
 HZ_PARTY_SITES HPS,
 PER_PERSON_NAMES_F PPNF  ,
 PO_HEADERS_ALL PHA  ,
 PO_LINES_ALL PLA ,
 PO_LINE_LOCATIONS_ALL PLLA  ,
 PO_DISTRIBUTIONS_ALL PDA  ,
 RCV_SHIPMENT_HEADERS RSH ,
 EGP_SYSTEM_ITEMS_B ESIB ,
 POZ_SUPPLIERS PS, 
 HR_ORGANIZATION_UNITS HOU 
WHERE
1=1
AND
HPS.OBJECT_VERSION_NUMBER=HP.OBJECT_VERSION_NUMBER
AND
PPNF.OBJECT_VERSION_NUMBER=HPS.OBJECT_VERSION_NUMBER
AND
PPNF.OBJECT_VERSION_NUMBER=HPS.OBJECT_VERSION_NUMBER
AND
PHA.OBJECT_VERSION_NUMBER=HPS.OBJECT_VERSION_NUMBER
AND
PLA.OBJECT_VERSION_NUMBER=HPS.OBJECT_VERSION_NUMBER
AND
PLLA.REQ_BU_ID=PLA.REQ_BU_ID
AND
PDA.LINE_LOCATION_ID=PLLA.LINE_LOCATION_ID
AND
RSH.OBJECT_VERSION_NUMBER=PDA.OBJECT_VERSION_NUMBER
AND
ESIB.OBJECT_VERSION_NUMBER=RSH.OBJECT_VERSION_NUMBER
AND
PS.PARTY_ID=HP.PARTY_ID
AND
HOU.OBJECT_VERSION_NUMBER=RSH.OBJECT_VERSION_NUMBER
