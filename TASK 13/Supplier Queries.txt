1.       Build a query for Inactive supplier List

SELECT PSV.VENDOR_NAME AS SupplierName,
HP.STATUS AS Status 
FROM
POZ_SUPPLIERS_V PSV
FULL JOIN 
HZ_PARTIES HP
ON PSV.PARTY_ID=HP.PARTY_ID
WHERE HP.STATUS='I'

3.       Get the Inactive Supplier Site List

SELECT 
A.STATUS,
B.PARTY_SITE_NAME
FROM 
HZ_PARTY_SITES A, 
POZ_SUPPLIER_SITES_V B
A.PARTY_SITE_ID=B.PARTY_SITE_ID
AND
A.STATUS IN ('I')

4.       Build a Query for the Supplier Type other than ‘SUPPLIER’

SELECT *
FROM POZ_SUPPLIERS WHERE VENDOR_TYPE_LOOKUP_CODE
NOT IN ('SUPPLIER')
 
 
5.       Build a Query for Suppliers those are not having Supplier Site
SELECT 
VENDOR_NAME, VENDOR_ID
FROM
POZ_SUPPLIERS_V 
WHERE VENDOR_ID IN 
(SELECT VENDOR_ID FROM POZ_SUPPLIERS_V
MINUS
SELECT VENDOR_ID FROM POZ_SUPPLIER_SITES_V)

6.       Build a Query for Supplier contact

SELECT PV.VENDOR_NAME,
PV.VENDOR_ID,
H.ADDRESS1,
H.CITY,
H.POSTAL_CODE,
H.STATE,
H.COUNTRY,
H.EMAIL_ADDRESS
FROM 
POZ_SUPPLIERS_V PV,
POZ_SUPPLIERS P,
HZ_PARTIES H
WHERE
PV.VENDOR_ID=P.VENDOR_ID
AND
H.PARTY_ID(+)=P.PARTY_ID


7.       Write a Query for Supplier and its site List those are assigned to Vision Corporation Business Unit

SELECT DISTINCT NAME,VENDOR_NAME,PARTY_SITE_NAME
FROM HR_ALL_ORGANIZATION_UNITS HRO
FULL JOIN POZ_SUPPLIERS_V POS 
ON HRO.OBJECT_VERSION_NUMBER = POS.OBJECT_VERSION_NUMBER
FULL JOIN HZ_PARTY_SITES HZP
ON HZP.PARTY_ID = POS.PARTY_ID
WHERE NAME = 'Vision Corporation'

8.       Write a query for the supplier’s having Purchase order value more than 1000$

SELECT VENDOR_NAME,SUM(UNIT_PRICE * QUANTITY) AS AMOUNT
FROM PO_LINES_ALL PLA FULL JOIN PO_HEADERS_ALL PHA ON PLA.PO_HEADER_ID=PHA.PO_HEADER_ID FULL JOIN POZ_SUPPLIERS_V PSV ON PSV.VENDOR_ID=PHA.VENDOR_ID
GROUP BY VENDOR_NAME
HAVING SUM(UNIT_PRICE * QUANTITY) >=1000

9.       List Supplier and its open PO counts

SELECT DISTINCT VENDOR_NAME,COUNT(DOCUMENT_STATUS) AS OPEN_COUNT
FROM PO_HEADERS_ALL POH FULL JOIN POZ_SUPPLIERS_V POS ON POH.VENDOR_ID = POS.VENDOR_ID 
WHERE DOCUMENT_STATUS = 'OPEN'
GROUP BY VENDOR_NAME 

10.   List of Partially Paid invoices with Supplier Name(Supplier, Business Unit , Invoice Number, Open Invoice Amount)

SELECT
HAOU.NAME AS BUINESS_UNIT,
AIA.INVOICE_NUM AS INVOICE_NUMBER,
(INVOICE_AMOUNT-AMOUNT_PAID) AS OPEN_INVOICE_AMT,
PSV.VENDOR_NAME AS SUPPLIER
FROM
AP_INVOICES_ALL AIA,
HR_ALL_ORGANIZATION_UNITS HAOU,
POZ_SUPPLIERS_V PSV
WHERE 
(INVOICE_AMOUNT-AMOUNT_PAID) > 0
AND
AIA.ORG_ID=HAOU.ORGANIZATION_ID
AND 
HAOU.OBJECT_VERSION_NUMBER=PSV.OBJECT_VERSION_NUMBER

11.   List the invoices which is having PO details

A.INVOICE_NUM,
A.PO_HEADER_ID 
FROM 
AP_INVOICES_ALL A, PO_HEADERS_ALL B 
WHERE 
B.PO_HEADER_ID= A.PO_HEADER_ID

12.   Write the Cancelled AP Invoice List between 1-Jan-2021 to 30-Jun-2021

SELECT INVOICE_NUM, APPROVAL_STATUS, TO_CHAR(CANCELLED_DATE, 'DD-MM-YYYY') AS CANCELLED_DATE 
FROM AP_INVOICES_ALL
WHERE APPROVAL_STATUS = 'CANCELLED'
AND TO_CHAR(CANCELLED_DATE, 'DD-MM-YYYY') BETWEEN  '01-01-2021'AND '30-06-2021'
